#!/bin/sh
content="$(wget --secure-protocol=TLSv1 --timeout=3 -t 1 -q -O - http://bj.ganji.com/piao/zz_北京南-天津/)"
if [ -z "$content" ];then
         notify-send  "Internet Connection Error."
         exit
fi
str="$(echo "$content"|grep ">\[转让\]"|sed 's#.*\(\[转让\].[^<]*\)<.*#\1#')"
  
notify-send "$str"
[root@localhost pub]#
